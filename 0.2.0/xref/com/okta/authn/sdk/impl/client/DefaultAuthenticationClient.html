257     private void handleResult(AuthenticationResponse authenticationResponse, AuthenticationStateHandler authenticationStateHandler) {
258         AuthenticationStatus status = authenticationResponse.getStatus();
259 
260          switch (status) {
261              case SUCCESS:
262                  authenticationStateHandler.handleSuccess(authenticationResponse);
263                  break;
264              case LOCKED_OUT:
265                  authenticationStateHandler.handleLockedOut(authenticationResponse);
266                  break;
267              case MFA_CHALLENGE:
268                  authenticationStateHandler.handleMfaChallenge(authenticationResponse);
269                  break;
270              case MFA_ENROLL:
271                  authenticationStateHandler.handleMfaEnroll(authenticationResponse);
272                  break;
273              case MFA_ENROLL_ACTIVATE:
274                  authenticationStateHandler.handleMfaEnrollActivate(authenticationResponse);
275                  break;
276              case MFA_REQUIRED:
277                  authenticationStateHandler.handleMfaRequired(authenticationResponse);
278                  break;
279              case PASSWORD_EXPIRED:
280                  authenticationStateHandler.handlePasswordExpired(authenticationResponse);
281                  break;
282              case PASSWORD_RESET:
283                  authenticationStateHandler.handlePasswordReset(authenticationResponse);
284                  break;
285              case PASSWORD_WARN:
286                  authenticationStateHandler.handlePasswordWarning(authenticationResponse);
287                  break;
288              case RECOVERY:
289                  authenticationStateHandler.handleRecovery(authenticationResponse);
290                  break;
291              case RECOVERY_CHALLENGE:
292                  authenticationStateHandler.handleRecoveryChallenge(authenticationResponse);
293                  break;
294              case UNAUTHENTICATED:
295                  authenticationStateHandler.handleUnauthenticated(authenticationResponse);
296                  break;
297              default:
298                  authenticationStateHandler.handleUnknown(authenticationResponse);
299          }
300     }
301 
302     private AuthenticationResponse doPost(String href, Resource request, AuthenticationStateHandler authenticationStateHandler) throws AuthenticationException {
303         try {
304             AuthenticationResponse authenticationResponse = getDataStore().create(href, request, AuthenticationResponse.class);
305             handleResult(authenticationResponse, authenticationStateHandler);
306             return authenticationResponse;
307         } catch (ResourceException e) {
308             translateException(e);
309             throw e; // above method should always throw
310         }
311     }
312 
313     private void translateException(ResourceException resourceException) throws AuthenticationException {
314 
315         String errorCode = resourceException.getCode();
316 
317         switch (errorCode) {
318             case AuthenticationFailureException.ERROR_CODE:
319                 throw new AuthenticationFailureException(resourceException);
320 
321             case CredentialsException.ERROR_CODE:
322                 throw new CredentialsException(resourceException);
323 
324             case FactorValidationException.ERROR_CODE:
325                 throw new FactorValidationException(resourceException);
326 
327             case InvalidAuthenticationStateException.ERROR_CODE:
328                 throw new InvalidAuthenticationStateException(resourceException);
329 
330             case InvalidRecoveryAnswerException.ERROR_CODE:
331                 throw new InvalidRecoveryAnswerException(resourceException);
332 
333             case InvalidTokenException.ERROR_CODE:
334                 throw new InvalidTokenException(resourceException);
335 
336             case InvalidUserException.ERROR_CODE:
337                 throw new InvalidUserException(resourceException);
338 
339             default:
340                 throw resourceException;
341         }
